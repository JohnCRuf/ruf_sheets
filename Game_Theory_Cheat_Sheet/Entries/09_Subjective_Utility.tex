\section{Subjective Utility}
\subsection{Basic Definitions}
States $s \in S$ (finite). Acts $f: S \rightarrow \mathbb{R}_+$, set of acts $F$. Acts are closed under addition and scalar multiplication.

\subsection{SEU Representation}
$\succeq$ has a subjective expected utility (SEU) rep $p \in \Delta(S)$ if
$f \succeq g \iff \sum_s p(s)f(s) \geq \sum_s p(s)g(s)$.
If $p$ and $q$ are both SEU reps for $\succeq$, then $p = q$.

\subsection{SEU Axioms}
\textbf{SEU Existence Theorem}  $\succeq$ on F has a SEU rep iff it is rational, monotonic, additive, non-trivial, and continuous.
Note that $\succeq$ must be rational. \\
 \textbf{1. Monotonicity} \\
$f(s) \geq g(s) \forall s \implies f \succeq g$ \\
 \textbf{2. Additivity} \\
$f \succeq g \iff f+h \succeq g+h \forall h$ \\
\textbf{3. Non-triviality} \\
$ \exists f,g, \ s.t. \ f \succ g$ \\
\textbf{4. Continuity} \\
If $f \succeq g \succeq h$, then $\exists \ \alpha \in [0,1]$ s.t. $\alpha f + (1-\alpha) h \sim g$
